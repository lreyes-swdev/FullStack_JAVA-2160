#
# Flex
#
# La propiedad CSS flex es una propiedad resumida que indica la capacidad de un elemento flexible
# para alterar sus dimensiones y llenar el espacio disponible.
# Los elementos flexibles pueden ser estirados para utilizar el espacio disponible
# proporcional a su factor de crecimiento flexible o su factor de contracción flexible para evitar desbordamiento.
#
# Un CONTENEDOR FLEXIBLE es la caja generada por un elemento con display: flex or inline-flex.
# Los elementos secundarios se denominan ELEMENTOS FLEXIBLES y se distribuyen utilizando el modelo de distribución flexible.
< div class="flex-container">
    < div class="flex-item">flex item 1< /div>
    < div class="flex-item">flex item 2< /div>
    < div class="flex-item">flex item 3< /div>
< /div>
.flex-container {
    display: flex | inline-flex;
    flex-direction: row | row-reverse | column | column-reverse;
    flex-wrap: nowrap | wrap | wrap-reverse;
    justify-content: flex-start | flex-end | center | space-between | space-around | space-evenly | ... ;
    align-items: flex-start | flex-end | center | baseline | stretch | ... ;
    align-content: flex-start | flex-end | center | space-between | space-around | space-evenly | stretch | baseline | ... ;
}
.flex-item {
    align-self: auto | flex-start | flex-end | center | baseline | stretch;
    order: 5;                /* default 0    */
    flex-grow: 4;            /* default 0    */
    flex-shrink: 3;          /* default 1    */
    flex-basis: 10% | auto;  /* default auto */
    flex: none | [ <'flex-grow'> <'flex-shrink'>? || <'flex-basis'> ]
}
