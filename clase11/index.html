# Animaciones en CSS con @keyframes

# @keyframes  ::  from + to
#flecha {
    color: brown;
    animation-name: flecha;
    animation-duration: 6s;
    animation-iteration-count: 2;
    animation-direction: reverse;
    animation-delay: 4s;
    position: absolute;
    left: 250px;
}

@keyframes flecha {
    from {
        color: blue;
        top: 250px;
    }
    to {
        color: brown;
        top: 50px;
    }
}

# @keyframes  ::  % + % + % + %
#uno, #tres {
    background-color: black;
    color: white;
    position: relative;
    top: 0px;
    animation-name: mover;
    animation-duration: 5s;
}

@keyframes mover {
    25% {
        top: 200px;
        left: 0px;
    }
    50% {
        left: 220px;
    }
    75% {
        top: 0px;
    }
    100% {
        left: 20px;
    }
}
